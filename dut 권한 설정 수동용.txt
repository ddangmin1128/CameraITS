# (필수) Android 13 이상 : CTS 인증 도구의 테스트 API 액세스 허용 
adb -s R5KL309P2KR shell am compat enable ALLOW_TEST_API_ACCESS com.android.cts.verifier

# Android 10 이상 : 앱에 보고서를 생성할 권한 허용
adb -s R5KL309P2KR shell appops set com.android.cts.verifier android:read_device_identifiers allow

# Android 11 이상 : 보고서 저장용 외부 최상위 디렉터리 액세스 허용
adb -s R5KL309P2KR shell appops set com.android.cts.verifier MANAGE_EXTERNAL_STORAGE 0

# Android 14 이상 : 앱에 화면 켜는 권한을허용
adb -s R5KL309P2KR shell appops set com.android.cts.verifier TURN_SCREEN_ON 0
